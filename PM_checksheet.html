▶ kv_store 테이블이 없을 때만 실행할 SQL 보기 
-- 이미 kv_store 테이블이 있다면 실행 불필요
-- (Supabase SQL Editor에서 "Key-Value JSON Store with RLS and Index" 쿼리 확인됨)

CREATE TABLE IF NOT EXISTS kv_store (
  key TEXT PRIMARY KEY,
  value JSONB NOT NULL DEFAULT '[]',
  updated_at TIMESTAMPTZ DEFAULT now()
);

ALTER TABLE kv_store ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Allow all access" ON kv_store
  FOR ALL USING (true) WITH CHECK (true);

CREATE INDEX IF NOT EXISTS idx_kv_store_updated ON kv_store(updated_at DESC);
